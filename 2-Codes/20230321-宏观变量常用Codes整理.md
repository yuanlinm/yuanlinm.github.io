```R
pvalue <- function(x, ...) {
  # 构建数据y的向量，和分组变量g
  y <- unlist(x)
  g <- factor(rep(1:length(x), times = sapply(x, length)))
  if (is.numeric(y)) {
    # 数值型变量，采用标准的t检验（两组比较）
    p <- oneway.test(y ~ g)$p.value
  } else {
    # 分类变量，此案用卡方检验
    p <- chisq.test(table(y, g))$p.value
  }
  # Format the p-value, using an HTML entity for the less-than sign.
  # The initial empty string places the output on the line below the variable label.
  c("", sub("<", "&lt;", format.pval(p, digits = 3, eps = 0.001)))
}
stats <- function(x, ...) {
  y <- unlist(x)
  g <- factor(rep(1:length(x), times = sapply(x, length)))
  if (is.numeric(y)) {
    s <- oneway.test(y ~ g)$statistic
  } else {
    s <- chisq.test(table(y, g))$statistic
  }
  c("", sprintf("%.4f", s))
}
table_1 <- table1(strata, labels,
  groupspan = c(1, 4, 1), overall = T, render.continuous = c(. = "Mean (SD)"),
  extra.col = list(`P-value` = pvalue, `Statistic` = stats)
)
```

```R
# 逻辑回归分析参数输出
log_info <- function(fit) {
  a <- as.data.frame(coef(summary(fit)))
  P.value <- format.pval(a$`Pr(>|z|)`[c(2)], eps = 0.001)
  OR <- round(exp(a$Estimate[c(2)]), 2)
  b <- as.data.frame(confint(fit))[c(2), ]
  CI95 <- paste("(", round(exp(b$`2.5 %`), 2), ", ", round(exp(b$`97.5 %`), 2), ")", sep = "")
  dat <- data.frame(P.value = c(P.value), OR = c(OR), CI95 = c(CI95))
  return(dat)
}
# Cox 回归分析参数输出
cox_info <- function(fit) {
  a <- as.data.frame(summary(fit)$coefficients)
  P.value <- format.pval(round(a$`Pr(>|z|)`[1], 4), eps = 0.001)
  HR <- round(a$`exp(coef)`, 2)[1]
  b <- as.data.frame(summary(fit)$conf.int)[1, ]
  CI95 <- paste("(", round((b$`lower .95`), 2), ", ", round((b$`upper .95`), 2), ")", sep = "")
  dat <- data.frame(HR = c(HR), CI95 = c(CI95), P.value = c(P.value), "Sig" = stars.pval(round(a$`Pr(>|z|)`[1], 4)))
  return(dat)
}
```

## **四、 自定义 ggplot2 的坐标轴位置**

```R
# do a function to move xaxis or yaxis
# parameters:
# data, for ggplot2
# x_begin, x_end, x_ticks, y_begin, y_end, y_ticks
# x_sep, y_sep
# x, y, for ggplot::aes
# x_ticks_sz, y_ticks_sz
# axis label font size: lb_font_size
# return a plot could be saved
# 请在标注区绘图
axisparameters <- data.frame(
  parameter = c("begin", "end", "ticks_num", "sep", "axis_position", "ticks_sz"),
  x_axis = c("", "", "", "", "", ""),
  y_axis = c("", "", "", "", "", "")
)
# axisparameters setting:
#       parameter x_axis y_axis
# 1         begin   15.0   -4.0
# 2           end  100.0    4.0
# 3     ticks_num   10.0    5.0
# 4           sep   10.0    2.0
# 5 axis_position    0.0   15.0
# 6      ticks_sz    0.1    0.5 # ticks_size
axisparameters$x_axis <- c(15, 100, 10, 10, 0, 0.1)
axisparameters$y_axis <- c(-4, 4, 5, 2, 15, 0.5)
# all done, have fun!
moveaxisfunc <- function(axisparameters, lb_font_size, data, x, y) {
  # constants
  xaxis_begin <- axisparameters[1, 2]
  xaxis_end <- axisparameters[2, 2]
  xtotal_ticks <- axisparameters[3, 2]
  x_sep <- axisparameters[4, 2]
  x_loc <- axisparameters[5, 2]
  x_ticks_sz <- axisparameters[6, 2]
  yaxis_begin <- axisparameters[1, 3]
  yaxis_end <- axisparameters[2, 3]
  ytotal_ticks <- axisparameters[3, 3]
  y_sep <- axisparameters[4, 3]
  y_loc <- axisparameters[5, 3]
  y_ticks_sz <- axisparameters[6, 3]
  # X
  xtick_frame <-
    data.frame(
      ticks = seq(xaxis_begin, xaxis_end, x_sep),
      zero = 0
    ) %>%
    subset(ticks != 0)
  xlab_frame <- data.frame(
    lab = seq(xaxis_begin, xaxis_end, x_sep),
    zero = 0
  ) %>%
    subset(lab != 0)
  xtick_sz <- (tail(xlab_frame$lab, 1) - xlab_frame$lab[1]) / 128
  # Y
  ytick_frame <-
    data.frame(
      ticks = seq(yaxis_begin, yaxis_end, y_sep),
      zero = 0
    ) %>%
    subset(ticks != 0)
  ylab_frame <- data.frame(
    lab = seq(yaxis_begin, yaxis_end, y_sep),
    zero = 0
  ) %>%
    subset(lab != 0)
  ytick_sz <- (tail(ylab_frame$lab, 1) - ylab_frame$lab[1]) / 128
  # PLOT ----
  plot1 <- ggplot(data, aes(x, y)) +
    ############################################################ 在这里绘图
    geom_point(col = "gray") +
    geom_smooth(method = "lm", size = 1.5, se = F, col = "#BB2C26") +




    ############################################################# CHART JUNK
    # y axis line
    geom_segment(
      x = y_loc, xend = y_loc,
      y = ylab_frame$lab[1], yend = tail(ylab_frame$lab, 1),
      size = 1
    ) +
    # y ticks
    geom_segment(
      data = ytick_frame,
      aes(
        x = y_loc, xend = y_loc + y_ticks_sz,
        y = ticks, yend = ticks
      ), size = 1
    ) +
    # x axis line
    geom_segment(
      y = x_loc, yend = x_loc,
      x = xaxis_begin, xend = tail(xlab_frame$lab, 1),
      size = 1
    ) +
    # x ticks
    geom_segment(
      data = xtick_frame,
      aes(
        x = ticks, xend = ticks,
        y = zero, yend = x_ticks_sz
      ), size = 1
    ) +
    # labels
    geom_text(
      data = xlab_frame, aes(x = lab, y = zero, label = lab),
      family = "Times New Roman", vjust = 1.5, size = lb_font_size
    ) +
    geom_text(
      data = ylab_frame, aes(x = y_loc - 1, y = lab, label = lab),
      family = "Times New Roman", hjust = 1.5, size = lb_font_size
    ) +
    # THE DATA POINT
    theme_classic() +
    theme(
      axis.text.x = element_blank(),
      axis.ticks.x = element_blank(),
      axis.line.x = element_blank(),
      axis.ticks.y = element_blank(),
      axis.line.y = element_blank(),
      axis.text.y = element_blank()
    )
  return(plot1)
}
save(axisparameters, moveaxisfunc, file = "code/moveaxisfunc.Rdata")
```
